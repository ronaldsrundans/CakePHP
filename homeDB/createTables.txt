CREATE TABLE CPU (
cpuid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
cpuman VARCHAR(30) NOT NULL,
cpumodel VARCHAR(30) NOT NULL,
cpuname VARCHAR(30) NOT NULL,
cpufreq VARCHAR(10),
cpucache VARCHAR(10),
cpusocket VARCHAR(10),
cpuyear VARCHAR(10),
cpucores VARCHAR(10),
cputhreads VARCHAR(10),
cpudate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE HDD (
hddid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
hddman VARCHAR(30) NOT NULL,
hddmodel VARCHAR(30) NOT NULL,
hddmem VARCHAR(10) NOT NULL,
hddstatus VARCHAR(30),
hdddate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE SSD (
ssdid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
ssdman VARCHAR(30) NOT NULL,
ssdmodel VARCHAR(30) NOT NULL,
ssdmem VARCHAR(10) NOT NULL,
ssdstatus VARCHAR(30),
ssddate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE RAM (
ramid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
ramman VARCHAR(30) NOT NULL,
ramname VARCHAR(30) NOT NULL,
ramfreq VARCHAR(10),
rammem VARCHAR(10),
ramtype VARCHAR(10),
ramsticks int(6),
ramdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE MOTHER (
motherid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
motherman VARCHAR(30) NOT NULL,
mothername VARCHAR(30) NOT NULL,
mothermaxram VARCHAR(10),
motherdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE OS (
osid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
osman VARCHAR(30) NOT NULL,
osname VARCHAR(30) NOT NULL,
osdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)
CREATE TABLE GPU (
gpuid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
gpuman VARCHAR(30) NOT NULL,
gpuname VARCHAR(30) NOT NULL,
gpumemfreq VARCHAR(10),
gpuvram VARCHAR(10),
gpuvramtype VARCHAR(10),
gpubus VARCHAR(10),
gputdp VARCHAR(10),
gpupsu VARCHAR(10),
gpucorefreq VARCHAR(10),
gpuyear VARCHAR(10),
gpudate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE PC (
pcid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
pccpuid int(6)UNSIGNED,
pcmotherid int(6)UNSIGNED,
pcgpuid int(6)UNSIGNED,
pcosid int(6)UNSIGNED,
pchddid int(6)UNSIGNED,
pcssdid int(6)UNSIGNED,
pcramid int(6)UNSIGNED,
pcpsu VARCHAR(30) NOT NULL,
pccomment VARCHAR(30),
pcdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (pccpuid) REFERENCES CPU(cpuid),
FOREIGN KEY (pcmotherid) REFERENCES MOTHER(motherid),
FOREIGN KEY (pcgpuid) REFERENCES GPU(gpuid),
FOREIGN KEY (pcosid) REFERENCES OS(osid),
FOREIGN KEY (pchddid) REFERENCES HDD(hddid),
FOREIGN KEY (pcramid) REFERENCES RAM(ramid),
FOREIGN KEY (pcssdid) REFERENCES SSD(ssdid)
)

CREATE TABLE GAME(
gameid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
gamename VARCHAR(30) NOT NULL,
gameyear VARCHAR(10),
gamedate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE TEST(
testid INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
testgameid int(6)UNSIGNED,
testpcid int(6)UNSIGNED,
testresult VARCHAR(30),
testfps int(6),
testdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (testpcid) REFERENCES PC(pcid),
FOREIGN KEY (testgameid) REFERENCES GAME(gameid)
)

CREATE TABLE FURMARK (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
fps VARCHAR(30) NOT NULL,
pts VARCHAR(30) NOT NULL,
width VARCHAR(4) NOT NULL,
height VARCHAR(4) NOT NULL,
pcid int(6)UNSIGNED,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (pcid) REFERENCES PC(id)
)
