CREATE TABLE CPU (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
man VARCHAR(30) NOT NULL,
name VARCHAR(30) NOT NULL,
freq VARCHAR(10),
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE MOTHER (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
man VARCHAR(30) NOT NULL,
name VARCHAR(30) NOT NULL,
ram VARCHAR(10),
freq VARCHAR(10),
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (cpuid) REFERENCES CPU(id)
)

CREATE TABLE OS (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
man VARCHAR(30) NOT NULL,
name VARCHAR(30) NOT NULL,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)
CREATE TABLE GPU (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
man VARCHAR(30) NOT NULL,
name VARCHAR(30) NOT NULL,
memfreq VARCHAR(10),
corefreq VARCHAR(10),
year VARCHAR(10),
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE PC (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
cpuid int(6)UNSIGNED,
motherid int(6)UNSIGNED,
gpuid int(6)UNSIGNED,
osid int(6)UNSIGNED,
hdd VARCHAR(30) NOT NULL,
ssd VARCHAR(30) NOT NULL,
psu VARCHAR(30) NOT NULL,
furmarkfps int(6)UNSIGNED,
furmarkscore int(6)UNSIGNED,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (cpuid) REFERENCES CPU(id),
FOREIGN KEY (motherid) REFERENCES MOTHER(id),
FOREIGN KEY (gpuid) REFERENCES GPU(id),
FOREIGN KEY (osid) REFERENCES OS(id)
)

CREATE TABLE GAME(
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(30) NOT NULL,
year VARCHAR(10),
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE TEST(
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
gameid int(6)UNSIGNED,
pcid int(6)UNSIGNED,
test VARCHAR(30),
fps int(6)UNSIGNED,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
FOREIGN KEY (pcid) REFERENCES PC(id),
FOREIGN KEY (gameid) REFERENCES GAME(id)
)
